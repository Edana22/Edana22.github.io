# 二维数组转置

```c
 1 #include<stdio.h> 
 2 #include<math.h>
 3 
 4 int main(){
 5     //二行三列矩阵，第一行数据为1,2,3；第二行数据为4,5,6 
 6     int arr[2][3] = {{1,2,3},{4,5,6}};
 7     int arrTran[3][2]; //转置后的矩阵为 3*2
 8     int i,j;
 9     
10     //按行输出arr 
11     printf("arr is :\n");
12     for(i = 0; i < 2; i++){
13         for(j = 0; j < 3; j++){
14             printf("    %d", arr[i][j]);
15         }
16         printf("\n");
17     }
18     
19     //转置就是把i、j调换 
20     printf("arrTran is : \n");
21     for(j = 0; j < 3; j++){
22         for(i = 0; i < 2; i++){
23             arrTran[j][i] = arr[i][j];
24             printf("    %d", arrTran[j][i]);
25         }
26         printf("\n");
27     }
28 }
```
